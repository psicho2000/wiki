    mit MX-Master verbinden (langer Druck auf den Umschalter 1,2,3)

Ggf. Gestenschalter > Gestenschalter > Benutzerdefiniert

    1: maximal
    2: mittel

Daumenradrichtung: Invertiert

AutoHotkey
    $WheelRight::FastForward()
    $WheelLeft::Rewind()

    FastForward() {
        WinGetActiveTitle, Title
        if (InStr(Title, "VLC media player")) {
            Send +{Right}
        } else if (InStr(Title, "Microsoft Edge") or InStr(Title, "Google Chrome")) {
            Send {Right}
        } else {
            Send {WheelRight}
        }
    }

    Rewind() {
        WinGetActiveTitle, Title
        if (InStr(Title, "VLC media player")) {
            Send +{Left}
        } else if (InStr(Title, "Microsoft Edge") or InStr(Title, "Google Chrome")) {
            Send {Left}
        } else {
            Send {WheelLeft}
        }  
    }
