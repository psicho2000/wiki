AutoHotkey
    WheelRight::FastForward()
    WheelLeft::Rewind()

    FastForward()
    {
        WinGetActiveTitle, Title
        IfInString, Title, VLC media player
        {
            Send +{Right}
        }
        IfInString, Title, Microsoft Edge
        {
            Send {Right}
        }
        IfInString, Title, Google Chrome
        {
            Send {Right}
        }
    }
    Rewind()
    {
        WinGetActiveTitle, Title
        IfInString, Title, VLC media player
        {
            Send +{Left}
        }
        IfInString, Title, Microsoft Edge
        {
            Send {Left}
        }
        IfInString, Title, Google Chrome
        {
            Send {Left}
        }
    }
